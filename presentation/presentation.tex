\begin{frame}\frametitle{Training dataset}
    \begin{columns}[T]
        \begin{column}{.5\textwidth}
            CityScapes dataset:
            \begin{itemize}
                \item 30 classes reduced to 7
                
                \item 5 000 annotated images with fine annotations
                
                \item images reduced from 2048x1024 to 256x128
            \end{itemize}
        \end{column}
        \begin{column}{.5\textwidth}

                    6 & Person \\
                    7 & Vehicle \\
                    \hline
                \end{tabular}
